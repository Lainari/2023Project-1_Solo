CREATE TABLE account (
nickname VARCHAR(10) NOT NULL,
userId VARCHAR(10) NOT NULL,
userPassword VARCHAR(10) NOT NULL,
isAdmin BOOLEAN NULL,
PRIMARY KEY (nickname)
);

CREATE TABLE board (
boardNo BIGINT NOT NULL AUTO_INCREMENT,
writer VARCHAR(10) NOT NULL,
title VARCHAR(20) NOT NULL,
content TEXT NOT NULL,
hit BIGINT NOT NULL,
writeDate DATE NOT NULL,
PRIMARY KEY (boardNo, writer),
CONSTRAINT FK_account_TO_board_1 FOREIGN KEY (writer)
REFERENCES account (nickname)
ON DELETE CASCADE
ON UPDATE CASCADE
);

CREATE TABLE comment (
boardNo BIGINT NOT NULL,
nickname VARCHAR(30) NOT NULL,
comment TEXT NOT NULL,
commentDate DATETIME NOT NULL,
PRIMARY KEY (boardNo, nickname),
CONSTRAINT FK_board_TO_comment_1 FOREIGN KEY (boardNo)
REFERENCES board (boardNo)
ON DELETE CASCADE,
CONSTRAINT FK_account_TO_comment_1 FOREIGN KEY (nickname)
REFERENCES account (nickname)
ON DELETE CASCADE
ON UPDATE CASCADE
);

CREATE TABLE image (
boardNo BIGINT NULL,
memberBoardNo BIGINT NULL,
fileName VARCHAR(50) NOT NULL,
CONSTRAINT FK_board_TO_image_1 FOREIGN KEY (boardNo)
REFERENCES board (boardNo)
ON DELETE CASCADE,
CONSTRAINT FK_memberOfTeam_TO_image_1 FOREIGN KEY (memberBoardNo)
REFERENCES memberOfTeam (memberBoardNo)
ON DELETE CASCADE
);

CREATE TABLE memberOfTeam (
memberBoardNo BIGINT NOT NULL AUTO_INCREMENT,
memberName VARCHAR(20) NOT NULL,
introduceContent TEXT NULL,
PRIMARY KEY (memberBoardNo)
);
